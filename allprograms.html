   private void drawStar(Graphics2D g2d, int x, int y, int radius1, int radius2, int numPoints)
   {
      // The angle between each point on the star
      double angle = Math.PI / numPoints;
      // Create a polygon representing the star
      GeneralPath path = new GeneralPath();
      // First point of the star
      path.moveTo(x + Math.cos(0) * radius1, y - Math.sin(0) * radius1);
      // Alternate between outer and inner points
      for (int i = 1; i < 2 * numPoints; i++)
      {
         double currentAngle = i * angle;
         double currentRadius = (i % 2 == 0) ? radius1 : radius2;
         path.lineTo(x + Math.cos(currentAngle) * currentRadius, y - Math.sin(currentAngle) * currentRadius);
      }
      path.closePath();
      g2d.fill(path);
   }

   public void paintStar(int[][] vertaxCoordinates, Graphics g)
   {
      // 检查输入的顶点坐标数组是否有效
      if (vertaxCoordinates == null || vertaxCoordinates.length < 3)
      {
         throw new IllegalArgumentException("至少需要三个顶点来绘制多边形");
      }
      
      // 设置画笔颜色为黄色
      g.setColor(Color.YELLOW);
      
      // 分配两个数组来存储x坐标和y坐标
      int[] xPoints = new int[vertaxCoordinates.length];
      int[] yPoints = new int[vertaxCoordinates.length];
      
      // 将坐标复制到xPoints和yPoints数组中
      for (int i = 0; i < vertaxCoordinates.length; i++)
      {
         xPoints[i] = vertaxCoordinates[i][0];
         yPoints[i] = vertaxCoordinates[i][1];
      }
      
      // 使用Graphics类的fillPolygon方法来绘制填充的多边形
      g.fillPolygon(xPoints, yPoints, vertaxCoordinates.length);
   }
   
   private int[][] calculateStarCoordinates(int centerX, int centerY, int outerRadius, int rotationAngleDegrees)
   {
      final int POINT_COUNT = 5;
      double angle = 2 * Math.PI / POINT_COUNT;
      double rotationAngle = Math.toRadians(rotationAngleDegrees);
      
      int[][] coordinates = new int[POINT_COUNT * 2][2];
      for (int i = 0; i < POINT_COUNT; i++)
      {
         // 外圈顶点
         coordinates[i * 2] = new int[]
         {
            (int) (centerX + outerRadius * Math.cos(i * angle - Math.PI / 2 + rotationAngle)),
            (int) (centerY + outerRadius * Math.sin(i * angle - Math.PI / 2 + rotationAngle))
         };
         // 内圈顶点
         coordinates[i * 2 + 1] = new int[]
         {
            (int) (centerX + outerRadius / 2.618 * Math.cos((i + 0.5) * angle - Math.PI / 2 + rotationAngle)),
            (int) (centerY + outerRadius / 2.618 * Math.sin((i + 0.5) * angle - Math.PI / 2 + rotationAngle))
         };
      }
      return coordinates;
   }

      frame.add(flagPanel); // 将NationalFlag添加到窗口

import java.io.File;
import java.io.FileNotFoundException;
import java.util.Scanner;
public class DataStatistics
{
   public static void main(String[] args)
   {
      if (args.length != 1)
      {
         System.err.println("Usage: java DataStatistics <file-path>");
         return;
      }
      
      Scanner scanner = null;
      try
      {
         scanner = new Scanner(new File(args[0]));
         
         int n; // total rows
         int row = 0, col = 0;
         float sum;
         String grades = "FDCBA";
         String[] headings = new String[9];
         float[][] nums = new float[48][]; // actual n is less than 48
         float[] means = new float[9];
         float[] stdvs = new float[9];
         
         while (scanner.hasNextLine())
         {
            String line = scanner.nextLine();
            String[] columns = line.split("\t");
            if (row == 0)
            {
               headings = columns;
               row++;
               continue;
            }
            
            // Initialize the current row array
            nums[row - 1] = new float[9];
            
            for (col = 0; col < columns.length; col++)
            {
               if (col == 1)
               {
                  String gradeStr = columns[col].trim();
                  nums[row - 1][col] = grades.indexOf(gradeStr);
               }
               else
               {
                  nums[row - 1][col] = Float.parseFloat(columns[col].trim());
               }
            }
            
            row++;
         }
         
         n = row - 1;
         
         // Calculate means
         for (int c = 0; c < 9; c++)
         {
            sum = 0;
            for (int r = 0; r < n; r++)
            {
               sum += nums[r][c];
            }
            means[c] = sum / n;
         }
         
         // Calculate standard deviations
         for (int c = 0; c < 9; c++)
         {
            sum = 0;
            for (int r = 0; r < n; r++)
            {
               sum += Math.pow(nums[r][c] - means[c], 2);
            }
            stdvs[c] = (float) Math.sqrt(sum / n);
         }
         
         // Print results
         System.out.printf("Mean of ID: %.2f\n", means[0]);
         System.out.printf("Mean of Grade: %.2f\n", means[1]);
         System.out.printf("Mean of Score: %.2f\n", means[2]);
         System.out.printf("Mean of Absence: %.2f\n", means[3]);
         System.out.printf("Mean of Time: %.2f\n", means[4]);
         System.out.printf("Mean of Start: %.2f\n", means[5]);
         System.out.printf("Mean of Sub: %.2f\n", means[6]);
         System.out.printf("Mean of Quiz: %.2f\n", means[7]);
         System.out.printf("Mean of Order: %.2f\n", means[8]);
         
         System.out.printf("Stdv of ID: %.2f\n", stdvs[0]);
         System.out.printf("Stdv of Grade: %.2f\n", stdvs[1]);
         System.out.printf("Stdv of Score: %.2f\n", stdvs[2]);
         System.out.printf("Stdv of Absence: %.2f\n", stdvs[3]);
         System.out.printf("Stdv of Time: %.2f\n", stdvs[4]);
         System.out.printf("Stdv of Start: %.2f\n", stdvs[5]);

         System.out.printf("Stdv of Quiz: %.2f\n", stdvs[7]);
         System.out.printf("Stdv of Order: %.2f\n", stdvs[8]);
         
      }

      {
         System.err.println("File not found: "+ e.getMessage());

         {

         }
      }
